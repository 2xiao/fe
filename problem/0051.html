var solveNQueens = function (n) {
	let res = [];
	let queens = new Array(n).fill(-1);

	const valid = (row, col) => {
		for (let i = 0; i < row; i++) {
			if (
				// Ê£ÄÊü•Âêå‰∏ÄÂàó‰∏äÊòØÂê¶ÊúâÂÖ∂‰ªñÁöáÂêé
				queens[i] == col ||
				// Ê£ÄÊü•Âú®Â∑¶‰∏äËá≥Âè≥‰∏ãÁöÑÂØπËßíÁ∫ø‰∏äÊòØÂê¶ÊúâÂÖ∂‰ªñÁöáÂêé
				queens[i] - i == col - row ||
				// Ê£ÄÊü•Âú®Â∑¶‰∏ãËá≥Âè≥‰∏äÁöÑÂØπËßíÁ∫ø‰∏äÊòØÂê¶ÊúâÂÖ∂‰ªñÁöáÂêé
				queens[i] + i == col + row
			) {
				return false;
			}
		}
		return true;
	};

	const backtrack = (row) => {
		if (row == n) {
			// ÊâæÂà∞‰∏Ä‰∏™ÊàêÂäüËß£
			res.push([...queens]);
			return;
		}

		for (let col = 0; col < n; col++) {
			if (valid(row, col)) {
				// Âú®ÂΩìÂâç‰ΩçÁΩÆÊîæÁΩÆÁöáÂêé
				queens[row] = col;
				// ÁªßÁª≠Â∞ùËØï‰∏ã‰∏ÄË°å
				backtrack(row + 1);
				// ÂõûÊ∫ØÔºåÊ∏ÖÁ©∫ÂΩìÂâçË°åÁä∂ÊÄÅ
				queens[row] = -1;
			}
		}
	};

	backtrack(0);
	return res.map((arr) =>
		arr.map((col) => '.'.repeat(col) + 'Q' + '.'.repeat(n - col - 1))
	);
};
